    #msg {

      position:fixed; inset:0; display:flex; align-items:center; justify-content:center;

      text-align:center; font-size:34px; z-index:6; background:rgba(0,0,0,.55);

      padding:30px;

    }

    #staticOverlay {

      position:fixed; inset:0; pointer-events:none; z-index:4;

      background:url("https://media.giphy.com/media/oEI9uBYSzLpBK/giphy.gif");

      opacity:0; mix-blend-mode:screen;

<div id="ui">Sigils: 0 / 8</div>

<div id="msg">Click to begin<br><br>WASD to move · Mouse to look<br>Collect 8 sigils. Avoid The Warden.</div>

<audio id="ambient" src="ambient.mp3" loop></audio>

<script type="module">

import * as THREE from "https://cdn.jsdelivr.net/npm/three@0.160.0/build/three.module.js";


let scene, camera, renderer;

let keys = {};

let started = false;

let gameOver = false;

let collected = 0;


const ui = document.getElementById("ui");

const msg = document.getElementById("msg");

const staticOverlay = document.getElementById("staticOverlay");


const ambient = document.getElementById("ambient");

const staticSound = document.getElementById("staticSound");

const jump = document.getElementById("jump");


let velocity = new THREE.Vector3();

let direction = new THREE.Vector3();


let yaw = 0;

let pitch = 0;


let sigils = [];

let warden;


init();

animate();


function init() {

  scene = new THREE.Scene();

  scene.background = new THREE.Color(0x030303);

  scene.fog = new THREE.Fog(0x030303, 15, 85);


  camera = new THREE.PerspectiveCamera(75, innerWidth / innerHeight, 0.1, 1000);

  camera.position.set(0, 2, 8);


  renderer = new THREE.WebGLRenderer({ antialias:true });

  renderer.setSize(innerWidth, innerHeight);

  document.body.appendChild(renderer.domElement);


  const floorGeo = new THREE.PlaneGeometry(220, 220);

  const floorMat = new THREE.MeshStandardMaterial({ color:0x111111 });

  const floor = new THREE.Mesh(floorGeo, floorMat);

  floor.rotation.x = -Math.PI / 2;

  scene.add(floor);


  const ambientLight = new THREE.AmbientLight(0x222222);

  scene.add(ambientLight);


  const flashlight = new THREE.SpotLight(0xffffff, 4, 45, Math.PI / 7, 0.5, 1);

  flashlight.position.set(0, 0, 0);

  camera.add(flashlight);

  flashlight.target.position.set(0, 0, -1);

  camera.add(flashlight.target);

  scene.add(camera);


  makeTrees();

  makeSigils();

  makeWarden();


  document.addEventListener("keydown", e => keys[e.key.toLowerCase()] = true);

  document.addEventListener("keyup", e => keys[e.key.toLowerCase()] = false);


  document.body.addEventListener("click", startGame);

  document.addEventListener("mousemove", lookAround);


  window.addEventListener("resize", () => {

    camera.aspect = innerWidth / innerHeight;

    camera.updateProjectionMatrix();

    renderer.setSize(innerWidth, innerHeight);

  });

}


function startGame() {

  if (started) return;


  started = true;

  msg.style.display = "none";

  document.body.requestPointerLock();


  ambient.volume = 0.45;

  staticSound.volume = 0;

  ambient.play();

  staticSound.play();

}


function lookAround(e) {

  if (!started || document.pointerLockElement !== document.body) return;


  yaw -= e.movementX * 0.002;

  pitch -= e.movementY * 0.002;

  pitch = Math.max(-1.3, Math.min(1.3, pitch));


  camera.rotation.order = "YXZ";

  camera.rotation.y = yaw;

  camera.rotation.x = pitch;

}


function makeTrees() {

  const trunkMat = new THREE.MeshStandardMaterial({ color:0x17100b });

  const leafMat = new THREE.MeshStandardMaterial({ color:0x061506 });


  for (let i = 0; i < 140; i++) {

    let x = Math.random() * 200 - 100;

    let z = Math.random() * 200 - 100;


    if (Math.abs(x) < 8 && Math.abs(z) < 8) continue;


    const trunk = new THREE.Mesh(

      new THREE.CylinderGeometry(0.35, 0.5, 7, 8),

      trunkMat

    );

    trunk.position.set(x, 3.5, z);

    scene.add(trunk);


    const leaves = new THREE.Mesh(

      new THREE.ConeGeometry(2.6, 8, 8),

      leafMat

    );

    leaves.position.set(x, 9, z);

    scene.add(leaves);

  }

}


function makeSigils() {

  const loader = new THREE.TextureLoader();


  for (let i = 0; i < 8; i++) {

    const canvas = document.createElement("canvas");

    canvas.width = 256;

    canvas.height = 256;

    const c = canvas.getContext("2d");


    c.fillStyle = "white";

    c.fillRect(55, 35, 146, 190);

    c.fillStyle = "black";

    c.font = "bold 90px Arial";

    c.fillText("✹", 82, 155);


    const texture = new THREE.CanvasTexture(canvas);

    const mat = new THREE.MeshBasicMaterial({ map:texture, transparent:true });

    const sigil = new THREE.Mesh(new THREE.PlaneGeometry(3, 3.8), mat);


    sigil.position.set(Math.random() * 160 - 80, 2.3, Math.random() * 160 - 80);

    sigil.userData.taken = false;


    scene.add(sigil);

    sigils.push(sigil);

  }

}


function makeWarden() {

  const texture = new THREE.TextureLoader().load("warden.png");

  const mat = new THREE.SpriteMaterial({ map:texture, transparent:true });


  warden = new THREE.Sprite(mat);

  warden.position.set(45, 3, 45);

  warden.scale.set(6, 8, 1);

  scene.add(warden);

}


function updatePlayer() {

  direction.set(0, 0, 0);


  if (keys["w"]) direction.z -= 1;

  if (keys["s"]) direction.z += 1;

  if (keys["a"]) direction.x -= 1;

  if (keys["d"]) direction.x += 1;


  direction.normalize();


  const speed = keys["shift"] ? 0.22 : 0.13;


  const forward = new THREE.Vector3();

  camera.getWorldDirection(forward);

  forward.y = 0;

  forward.normalize();


  const right = new THREE.Vector3();

  right.crossVectors(forward, camera.up).normalize();


  camera.position.addScaledVector(forward, -direction.z * speed);

  camera.position.addScaledVector(right, direction.x * speed);


  camera.position.x = THREE.MathUtils.clamp(camera.position.x, -105, 105);

  camera.position.z = THREE.MathUtils.clamp(camera.position.z, -105, 105);

  camera.position.y = 2;

}


function updateWarden() {

  const playerPos = camera.position.clone();

  const wardenPos = warden.position.clone();


  const distance = playerPos.distanceTo(wardenPos);


  const dir = playerPos.sub(wardenPos).normalize();

  const speed = 0.025 + collected * 0.008;


  warden.position.addScaledVector(dir, speed);

  warden.position.y = 3;


  const staticAmount = Math.max(0, Math.min(0.9, 1 - distance / 45));

  staticOverlay.style.opacity = staticAmount;

  staticSound.volume = staticAmount;


  if (distance < 3.2) {

    loseGame();

  }

}


function updateSigils() {

  sigils.forEach(sigil => {

    if (sigil.userData.taken) return;


    sigil.lookAt(camera.position);


    const d = camera.position.distanceTo(sigil.position);

    if (d < 3) {

      sigil.userData.taken = true;

      sigil.visible = false;

      collected++;

      ui.innerHTML = `Sigils: ${collected} / 8`;


      if (collected >= 8) winGame();

    }

  });

}


function loseGame() {

  gameOver = true;

  ambient.pause();

  staticSound.pause();

  jump.play();


  msg.style.display = "flex";

  msg.innerHTML = "THE WARDEN FOUND YOU";

  document.exitPointerLock();

}


function winGame() {

  gameOver = true;

  ambient.pause();

  staticSound.pause();


  msg.style.display = "flex";

  msg.innerHTML = "You broke the signal.<br>You survived.";

  document.exitPointerLock();

}


function animate() {

  requestAnimationFrame(animate);


  if (started && !gameOver) {

    updatePlayer();

    updateWarden();

    updateSigils();

  }


  renderer.render(scene, camera);

}
